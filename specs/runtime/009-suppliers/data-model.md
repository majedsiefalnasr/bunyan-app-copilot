/**
 * Scope: visibility based on actor.
 * - Admin: sees all statuses (including pending/suspended)
 * - Contractor: sees verified + own profile
 * - Guest/others: sees only verified
 */
public function scopeVisibleTo(Builder $query, ?User $actor): Builder
{
    if ($actor?->hasEnumRole(\App\Enums\UserRole::ADMIN)) {
        return $query; // Admin sees all
    }

    if ($actor?->hasEnumRole(\App\Enums\UserRole::CONTRACTOR)) {
        return $query->where(function (Builder $q) use ($actor) {
            $q->where('verification_status', SupplierVerificationStatus::Verified->value)
              ->orWhere('user_id', $actor->id);
        });
    }

    // Guest / other roles: verified only
    return $query->where('verification_status', SupplierVerificationStatus::Verified->value);
}
